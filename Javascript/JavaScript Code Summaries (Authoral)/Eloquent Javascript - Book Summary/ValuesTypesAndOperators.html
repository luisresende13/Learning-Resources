Summary 1 - Eloquent Javascript

Values, Types and Operators

JavaScript Code:

// Back slash "\" to escape code characters like " and \ (double bracktes and back slash).

var text = "A newline character is written like \"\\n\"." console.log(text)


// Use Backtick-quoted strings `` and ${} to insert a variable's value into a string.

var text = 'Variable'; var variable = 100

console.log(`${text} = ${variable}`)


// Unary operator "type of" to return the data type of a value

console.log('data type of 4.5: ' + typeof 4.5)


// NaN != NaN ?

console.log("NaN==NaN?: " + String(NaN==NaN))

console.log("NaN===NaN?: " + String(NaN===NaN))


// Ternary operator (three values): (Condition ? value_if_true : else_vale)

a = 10; b = 20

console.log(`O maior valor entre ${a} e ${b} é: ${ a>b ? a : b }`)


// Automatic Type Conversions

console.log("8 * null: "+String(8 * null))

console.log("\"5\" - 1: "+String("5" - 1))

console.log("\"5\" + 1: "+String("5" + 1))
